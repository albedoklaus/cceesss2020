\begin{table}[]
\begin{tabular}[c|cc]
  $n$ & $N_\epsilon$ & $\frac{1}{\epsilon}$ \\\hline
  $0$ & $1$ & $1$ \\
  $1$ & $5$ & $3$ \\
  $2$ & $5^2$ & $3^2$ \\
  $3$ & $5^3$ & $3^3$ \\
  \vdots & \vdots & \vdots \\
  $n$ & $5^n$ & $3^n$ \\
\end{tabular}
\end{table}

$$ \Rightarrow d = \lim_{n \rightarrow \infty} \frac{n \log(5)}{n \log(3)} = 1.465$$
